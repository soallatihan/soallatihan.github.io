DIKETAHUI

n(A ∪ B) = 32 siswa

IPA = n(A) = 7 siswa

Matematika = n(B) = 9 siswa

Gemar keduanya = n(A ∩ B) = 5 siswa

PENYELESAIAN

n(A ∪ B) = n(A) + n(B) - n(A ∩ B) + n(A' ∩ B')

32 = 7 + 9 - 5 + n(A' ∩ B')

32 = 16 - 5 + n(A' ∩ B')

n(A' ∩ B') = 32 - 11

n(A' ∩ B') = 21 siswa


Untuk membuktikan;

n(A ∪ B) = n(A) + n(B) - n(A ∩ B) + n(A' ∩ B')

32 = 7 + 9 - n(A ∩ B) + 21

n(A ∩ B) = 37 - 32

n(A ∩ B) = 5 siswa
