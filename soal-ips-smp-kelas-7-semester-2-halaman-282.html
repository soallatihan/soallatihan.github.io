282°F = °C

282°F = 9/5 × t + 32°C

282°F = 9/5 × 282 + 32°C

282°F = 2.570/5°C

282°F = 514°C


282°F = K

282°F = 9/5 × (t-273) + 32K

282°F = 9/5 × (282-273) + 32K

282°F = 9/5 × 9 + 32K

282°F = 113/5K

282°F = 22,6K


282°F = °R

282°F = 9/4 × t + 32°R

282°F = 9/4 × 282 + 32°R

282°F = 2.570/4°R

282°F = 642,5°R

contoh : • a(b.c) = a.b + a.c

• a(b.c) = (a.b)c

• (a + b) + c = a + (b + c)
